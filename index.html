  private readonly hereMapGeocoder: HereMapGeocoder;

    return this.hereMapGeocoder
      .geocode(address)
      .catch((e) => {
        return this.googleGeocoder.geocode(address)
      })
  }
}

function geocode(addresses: string[], preference?: Preference): Promise<(Coords | null)[]> {
  let geocoder = ((preference == null) || (preference == Preference.Google))
    ? new GoogleAndHereMapGeocoder()
    : new HereMapAndGoogleGeocoder();
    
  return Promise.all(addresses.map(geocoder.geocode));
}

  private readonly googleGeocoder: GoogleGeocoder;
  private readonly hereMapGeocoder: HereMapGeocoder;

  public async geocode(address: string, preference?: Preference): Promise<(Coords | null)> {
    let [primary, secondary] = ((preference == null) || (preference == Preference.Google))
